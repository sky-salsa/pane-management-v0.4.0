    export function relativeTime(isoTimestamp: string | null): string {
      if (!isoTimestamp) return "unknown";
      const now = Date.now();
      const then = new Date(isoTimestamp).getTime();
      const diffMs = now - then;
      if (diffMs < 0) return "just now";
      const minutes = Math.floor(diffMs / 60_000);
      if (minutes < 1) return "just now";
      if (minutes < 60) return `${minutes}m ago`;
      const hours = Math.floor(minutes / 60);
      if (hours < 24) return `${hours}h ago`;
      const days = Math.floor(hours / 24);
      if (days < 30) return `${days}d ago`;
      return new Date(isoTimestamp).toLocaleDateString();
    }

    export function formatDuration(firstTimestamp: string | null, lastTimestamp: string | null): string {
      if (!firstTimestamp || !lastTimestamp) return "-";
      const ms = new Date(lastTimestamp).getTime() - new Date(firstTimestamp).getTime();
      if (ms < 0) return "-";
      const minutes = Math.floor(ms / 60000);
      const hours = Math.floor(minutes / 60);
      if (hours > 0) return `${hours}h ${minutes % 60}m`;
      if (minutes > 0) return `${minutes}m`;
      return "<1m";
    }
    ```

      projects: ProjectWithMeta[];
      selectedTmuxSession: string | null;  // currently viewed tmux session
      selectedTmuxWindow: number | null;    // currently viewed tmux window
      tmuxSessions: TmuxSession[];
      tmuxWindows: TmuxWindow[];
      tmuxPanes: TmuxPane[];
      paneAssignments: Record<string, string>;  // pane_index -> encoded_project

    }
    ```

    3. **Define AppContextValue interface** exposing state + action functions:
    ```typescript
    interface AppContextValue {
      state: AppState;
      // Selection
      selectTmuxSession: (name: string) => void;
      selectTmuxWindow: (index: number) => void;
      // Refresh
      refreshProjects: () => void;
      refreshTmuxState: () => void;
      refreshPanePresets: () => void;
      // Getters (derived)
      projectsByTier: (tier: ProjectTier) => ProjectWithMeta[];
      getProjectMeta: (encodedName: string) => ProjectMeta;
    }
    ```

    4. **Implement AppProvider component:**
    - Use `createStore<AppState>` for the main state
    - On mount: load projects via `listProjects()`, load project meta via `getAllProjectMeta()`, merge them into `ProjectWithMeta[]` (for each project, look up meta by encoded_name; if no meta exists, use default: `{ display_name: null, tier: "active", bound_session: null }`)
    - On mount: load tmux sessions via `listTmuxSessions()`, auto-select "workspace" session if it exists (or first attached session), then query windows for that session
    - **On mount: load pane assignments via `getPaneAssignments()`** and store in `state.paneAssignments`. This is critical for DASH-04 persistence -- without it, pane assignments would be lost after restart.
    - **On mount: load pane presets via `getPanePresets()`** and store in `state.panePresets`. This ensures PanePresetPicker (Plan 04) has data immediately available without needing to fetch its own.
    - On mount: start active session polling with `setInterval(pollActiveSessions, 5000)` (same pattern as DevPanel)
    - Set up tmux state polling: `setInterval(refreshTmuxState, 3000)` -- polls tmux state for selected session/window. Do NOT poll faster than 3 seconds (per research Pitfall 2).
    - Listen for `"session-changed"` Tauri event (from file watcher) and refresh projects when received
    - `refreshProjects`: re-fetch listProjects + getAllProjectMeta, merge, update store
    - `refreshTmuxState`: if selectedTmuxSession and selectedTmuxWindow are set, call `getTmuxState`, update tmuxSessions/tmuxWindows/tmuxPanes in store. Also re-fetch `getPaneAssignments()` and update `state.paneAssignments` to stay in sync with any changes made by other parts of the app.
    - `refreshPanePresets`: re-fetch `getPanePresets()` and update `state.panePresets`
    - `selectTmuxSession(name)`: update selectedTmuxSession, fetch windows for that session, auto-select first window, fetch panes
    - `selectTmuxWindow(index)`: update selectedTmuxWindow, fetch panes for that window
    - `projectsByTier(tier)`: filter state.projects by meta.tier === tier
    - `getProjectMeta(encodedName)`: find in state.projects or return default
    - Clean up intervals on unmount

    5. **Export `useApp()` hook:**
    ```typescript
    export function useApp(): AppContextValue {
      const ctx = useContext(AppContext);
      if (!ctx) throw new Error("useApp must be used within AppProvider");
      return ctx;
    }
    ```

    **Important implementation notes:**
    - Use `batch()` when updating multiple store fields at once to prevent unnecessary re-renders

    - The `ProjectWithMeta` merge logic: take the `ProjectInfo[]` from listProjects, map each to `ProjectWithMeta` by looking up `meta[project.encoded_name]` or using the default ProjectMeta
