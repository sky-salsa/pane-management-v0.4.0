    **Implement watcher.rs:**

    Create `pub async fn start_watcher(app_handle: tauri::AppHandle) -> Result<notify::RecommendedWatcher, String>` that:

    1. Creates a `notify::RecommendedWatcher` watching `~/.claude/projects/` recursively.
    2. Uses a tokio mpsc channel to bridge from notify's sync callback to async processing.
    3. Implements debouncing: collect events for 3 seconds before processing. Use a tokio task that:
       - Receives events into a buffer
       - After 3 seconds of quiet (no new events), processes the batch
       - Filters to only `.jsonl` file creates/modifies (ignore directory events, non-jsonl files)
       - Deduplicates paths in the batch
    4. When a batch fires, emit a Tauri event `"session-changed"` with the list of changed file paths as payload.
    5. Per locked decision: watcher runs always in background. The frontend decides whether to react.

    ```rust
    use notify::{Watcher, RecursiveMode, RecommendedWatcher, Event, EventKind};
    use tokio::sync::mpsc;
    use tokio::time::{sleep, Duration, Instant};
    use std::collections::HashSet;

    pub async fn start_watcher(
        app_handle: tauri::AppHandle,
    ) -> Result<RecommendedWatcher, String> {
        let (tx, mut rx) = mpsc::channel::<Event>(256);

        let mut watcher = RecommendedWatcher::new(
            move |res: Result<Event, notify::Error>| {
                if let Ok(event) = res {
                    let _ = tx.blocking_send(event);
                }
            },
            notify::Config::default(),
        ).map_err(|e| format!("Failed to create watcher: {}", e))?;

        let claude_projects = dirs::home_dir()
            .ok_or("Cannot find home directory")?
            .join(".claude")
            .join("projects");

        if claude_projects.exists() {
            watcher.watch(&claude_projects, RecursiveMode::Recursive)
                .map_err(|e| format!("Failed to watch directory: {}", e))?;
        }

        // Debounced event processing
        tokio::spawn(async move {
            let mut pending: HashSet<String> = HashSet::new();
            let mut last_event = Instant::now();
            let debounce_duration = Duration::from_secs(3);

            loop {
                tokio::select! {
                    event = rx.recv() => {
                        match event {
                            Some(e) => {
                                for path in &e.paths {
                                    if path.extension().map(|ext| ext == "jsonl").unwrap_or(false) {
                                        pending.insert(path.to_string_lossy().to_string());
                                    }
                                }
                                last_event = Instant::now();
                            }
                            None => break, // Channel closed
                        }
                    }
                    _ = sleep(Duration::from_millis(500)) => {
                        if !pending.is_empty() && last_event.elapsed() >= debounce_duration {
                            let paths: Vec<String> = pending.drain().collect();
                            let _ = app_handle.emit("session-changed", &paths);
                        }
                    }
                }
            }
        });

        Ok(watcher)
    }
    ```

       **CRITICAL per locked decision (visibility-gated refresh):** When the event fires, check `document.visibilityState === 'visible'` BEFORE calling refetch(). If the document is hidden (app minimized or tab not focused), do NOT refetch. Instead, set a flag (e.g., `needsRefresh`) and check it on a `visibilitychange` event listener to refetch when the app becomes visible again. This prevents wasted rendering when minimized.
    5. Show a status line: "Watching for changes..." with a timestamp of last refresh.
    6. Show total counts: "{N} projects, {M} total sessions"

    ```tsx
    import { createSignal, createResource, For, Show, onMount, onCleanup } from 'solid-js';
    import { listen } from '@tauri-apps/api/event';
    import { listProjects, listSessions } from '../lib/tauri-commands';
    import type { ProjectInfo, SessionInfo } from '../lib/types';

    export function DevPanel() {
      const [refreshKey, setRefreshKey] = createSignal(0);
      const [projects, { refetch }] = createResource(refreshKey, listProjects);
      const [selectedProject, setSelectedProject] = createSignal<string | null>(null);
      const [sessions, setSessions] = createSignal<SessionInfo[]>([]);
      const [lastRefresh, setLastRefresh] = createSignal(new Date().toISOString());
      let needsRefresh = false;

      onMount(async () => {
        // Visibility-gated refresh: only refetch when app is visible
        const unlisten = await listen<string[]>('session-changed', () => {
          if (document.visibilityState === 'visible') {
            refetch();
            setLastRefresh(new Date().toISOString());
          } else {
            needsRefresh = true;
          }
        });

        // Refetch when becoming visible if we missed events while hidden
        const onVisibilityChange = () => {
          if (document.visibilityState === 'visible' && needsRefresh) {
            needsRefresh = false;
            refetch();
            setLastRefresh(new Date().toISOString());
          }
        };
        document.addEventListener('visibilitychange', onVisibilityChange);

        onCleanup(() => {
          unlisten();
          document.removeEventListener('visibilitychange', onVisibilityChange);
        });
      });

      async function showSessions(encodedName: string) {
        setSelectedProject(encodedName);
        const result = await listSessions(encodedName);
        setSessions(result);
      }

      // ... render project list, session detail, status line
    }
    ```
