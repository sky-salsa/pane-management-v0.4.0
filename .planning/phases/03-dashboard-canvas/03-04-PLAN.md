    ```typescript
    /**
     * Convert a Windows path (e.g., C:\Users\USERNAME\project) to a WSL path (/mnt/c/Users/USERNAME/project).
     * Used when sending commands to tmux panes running inside WSL.
     */
    export function toWslPath(winPath: string): string {
      const p = winPath.replace(/\\/g, "/");
      if (p.length >= 2 && p[1] === ":") {
        const drive = p[0].toLowerCase();
        return `/mnt/${drive}${p.slice(2)}`;
      }
      return p;
    }
    ```

    5. Add an `onDragEnd` callback. The best approach with @thisbeyond/solid-dnd is to use `DragDropProvider`'s `onDragEnd` prop:

    ```tsx
    async function handleDragEnd({ draggable, droppable }: DragEvent) {
      if (!draggable || !droppable) return;

      const encodedProject = draggable.id as string;
      const paneIndex = parseInt(droppable.id as string, 10);

      // Look up the project for path info
      const project = state.projects.find(p => p.encoded_name === encodedProject);
      if (!project) return;

      // 1. Persist the pane assignment
      await setPaneAssignment(paneIndex, encodedProject);

      // 2. Determine the command to send to the pane
      const wslPath = toWslPath(project.actual_path);
      const sessionId = project.meta.bound_session || "";
      const command = sessionId
        ? `cd "${wslPath}" && claude -r ${sessionId}`
        : `cd "${wslPath}" && claude -r`;

      // 3. Send the command to the tmux pane
      const session = state.selectedTmuxSession;
      const window = state.selectedTmuxWindow;
      if (session != null && window != null) {
        await sendToPane(session, window, paneIndex, command);
      }

      // 4. Refresh tmux state to show the updated pane
      refreshTmuxState();
      refreshProjects(); // refresh to pick up assignment changes
    }
    ```
